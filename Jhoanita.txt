    <script>
        // Función para hacer que los corazones giren
        function girarCorazones() {
            const hearts = document.querySelectorAll('.heart');
            hearts.forEach(heart => {
                heart.style.transform = `rotate(${Math.random() * 360}deg)`;
            });
        }

        // Hacer que los corazones giren cada segundo
        setInterval(girarCorazones, 1000);

        // Función que muestra el mensaje cuando se encuentra el corazón verde
        function mostrarMensaje() {
            document.getElementById('message').style.display = 'block';
        }
